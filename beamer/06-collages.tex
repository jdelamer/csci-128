\begin{frame}{Bounds Checking: Why It Matters}

\begin{frame}[fragile]{Pasting an Image: Manual Method}
    \begin{lstlisting}
def paste_image(canvas, source, x_offset, y_offset):
    canvas_pixels = canvas.load()
    source_pixels = source.load()
    src_width, src_height = source.size

    for x in range(src_width):
        for y in range(src_height):
            # Calculate where this pixel goes on canvas
            dest_x = x_offset + x  # Shift by offset
            dest_y = y_offset + y

            # Check bounds (don't go off canvas!)
            if (0 <= dest_x < canvas.size[0] and
                0 <= dest_y < canvas.size[1]):
                # Copy the pixel color
                canvas_pixels[dest_x, dest_y] = source_pixels[x, y]
    return canvas
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Generalized Grid Collage}
    \begin{lstlisting}
def create_grid_collage(images, rows, cols, cell_size):
    # Calculate total canvas size
    canvas_width = cols * cell_size
    canvas_height = rows * cell_size
    canvas = Image.new("RGB", (canvas_width, canvas_height))
    for i, img in enumerate(images):
        if i >= rows * cols:  # Don't exceed grid capacity
            break
        # Calculate grid position using math trick
        row = i // cols  # Which row? (integer division)
        col = i % cols   # Which column? (remainder)
        x = col * cell_size  # Pixel x position
        y = row * cell_size  # Pixel y position
        # Scale and paste
        scaled = img.resize((cell_size, cell_size))
        canvas.paste(scaled, (x, y))
    return canvas
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Grid Collage with Spacing}

def create_grid_with_spacing(images, rows, cols, cell_size, spacing):
    canvas_width = cols * cell_size + (cols + 1) * spacing
    canvas_height = rows * cell_size + (rows + 1) * spacing
    canvas = Image.new("RGB", (canvas_width, canvas_height),
                       (240, 240, 240))  # Light gray

    for i, img in enumerate(images):
        if i >= rows * cols:
            break

        row = i // cols
        col = i % cols
        x = (col + 1) * spacing + col * cell_size
        y = (row + 1) * spacing + row * cell_size
        scaled = img.resize((cell_size, cell_size))
        canvas.paste(scaled, (x, y))

    return canvas
    \end{lstlisting}

\begin{frame}{Blending Images}
    \begin{columns}
    \begin{column}{0.55\textwidth}
        \textbf{Blending combines two images:}
        \begin{itemize}
            \item Both images contribute
            \item Creates "double exposure"
            \item Controlled by alpha ($\alpha$)
        \end{itemize}

        \vspace{0.5em}

        \textbf{Formula per pixel:}
        \begin{center}
            $\text{result} = \alpha \times \text{img1} + (1-\alpha) \times \text{img2}$
        \end{center}

        \vspace{0.5em}

        \textbf{Understanding $\alpha$:}
        \begin{itemize}\small
            \item $\alpha = 0.0$: Only img2
            \item $\alpha = 0.5$: Equal blend
            \item $\alpha = 1.0$: Only img1
        \end{itemize}
    \end{column}

\begin{frame}[fragile]{Implementing Image Blending}
    \begin{lstlisting}
def blend_images(img1, img2, alpha):
    width, height = img1.size # Ensure same size
    img2 = img2.resize((width, height))

    result = Image.new("RGB", (width, height))
    pix1 = img1.load()
    pix2 = img2.load()
    result_pix = result.load()

    for x in range(width):
        for y in range(height):
            r1, g1, b1 = pix1[x, y]
            r2, g2, b2 = pix2[x, y]
            r = int(alpha * r1 + (1 - alpha) * r2)
            g = int(alpha * g1 + (1 - alpha) * g2)
            b = int(alpha * b1 + (1 - alpha) * b2)
            result_pix[x, y] = (r, g, b)

    return result
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Simple Chromakey Implementation}
    \begin{lstlisting}
from PIL import Image

def chromakey(foreground, background, key_color,
              threshold=50):
    fg_pixels = foreground.load()
    bg_pixels = background.load()
    result = Image.new("RGB", foreground.size)
    result_pixels = result.load()

    for x in range(foreground.size[0]):
        for y in range(foreground.size[1]):
            r, g, b = fg_pixels[x, y]

            # Calculate "color distance" to key color
            # (How different is this pixel from green?)
            distance = abs(r - key_color[0]) + abs(g - key_color[1]) + abs(b - key_color[2])

            if distance < threshold:  # Close to green?
                result_pixels[x, y] = bg_pixels[x, y]  # Use background
            else:  # Not green?
                result_pixels[x, y] = (r, g, b)  # Use foreground

    return result
    \end{lstlisting}
\end{frame}

# Ensure same size
background = background.resize(foreground.size)

# Apply chromakey
# Green screen color (approximately)
green = (0, 255, 0)
result = chromakey(foreground, background, green,
                   threshold=100)

result.save("composited.jpg")
result.show()
    \end{lstlisting}

    \vspace{1em}

    \textbf{Note:} Threshold controls how close to green is replaced
\end{frame}

\begin{frame}{Common Chromakey Problems \& Solutions}
    \textbf{Challenges with simple chromakey:}

    \vspace{1em}

    \begin{itemize}
        \item \textbf{Hard edges} → Looks like a cutout, not realistic
        \item \textbf{Shadows on green screen} → Creates darker greens
        \item \textbf{Green spill on subject} → Green reflects onto person
        \item \textbf{Not all greens match} → Lighting variations
    \end{itemize}

    \vspace{1em}

    \textbf{Professional improvements:}
    \begin{itemize}
        \item Use HSV color space instead of RGB (better color matching)
        \item Add edge feathering (gradual transparency for smooth edges)
        \item Adjust threshold per-region (adapt to lighting)
        \item Color correction for spill (remove green tint)
    \end{itemize}

    \vspace{1em}

    \textit{You've learned the foundation - Hollywood uses these same principles!}
\end{frame}

\begin{frame}{Overlaying with Transparency}
    \begin{columns}
    \begin{column}{0.5\textwidth}
        \textbf{Some images have transparent regions:}

        \begin{itemize}
            \item PNG format supports alpha channel
            \item Alpha = 0: Fully transparent
            \item Alpha = 255: Fully opaque
            \item In between: Partially transparent
        \end{itemize}

        \vspace{0.5em}

        \textbf{Mode "RGBA":}
        \begin{itemize}
            \item R = Red (0-255)
            \item G = Green (0-255)
            \item B = Blue (0-255)
            \item A = Alpha (0-255)
        \end{itemize}
    \end{column}

\begin{frame}[fragile]{Pasting with Transparency}
    \begin{lstlisting}
from PIL import Image

# Load base image
canvas = Image.open("background.jpg")

# Load image with transparency (PNG)
overlay = Image.open("logo.png")

# Paste using the alpha channel as a mask
# The third parameter tells it to use transparency
canvas.paste(overlay, (100, 50), overlay)

canvas.save("with_overlay.jpg")
canvas.show()
    \end{lstlisting}

    \vspace{1em}

    \textbf{The third parameter (overlay) uses its alpha channel as a mask}

    \vspace{0.5em}

    Transparent pixels won't overwrite the canvas!
\end{frame}

\begin{frame}{Best Practices for Compositions}
    \textbf{Tips for great results:}

    \vspace{1em}

    \begin{enumerate}
        \item \textbf{Plan your layout first}
            \begin{itemize}
                \item Sketch on paper
                \item Calculate dimensions

        \item \textbf{Consider aspect ratios}
            \begin{itemize}

                \item Or embrace different shapes
            \end{itemize}

        \item \textbf{Use consistent spacing}

                \item Makes it look professional
            \end{itemize}

        \item \textbf{Pay attention to resolution}
            \begin{itemize}
                \item Don't scale up too much (pixelation)
                \item Start with high-quality sources
            \end{itemize}

        \item \textbf{Save in appropriate format}
            \begin{itemize}
                \item JPEG for photos, PNG for graphics
            \end{itemize}
    \end{enumerate}

\begin{frame}[fragile]{Performance Considerations}
    \textbf{Working with multiple images:}

    \vspace{1em}

    \begin{itemize}
        \item Multiple large images use lots of memory
        \item Scale images down before combining when possible
        \item Process images one at a time
        \item Close/delete images you're done with
        \item Test with small images first
    \end{itemize}

    \vspace{1em}

    \begin{lstlisting}
# Good practice
img = Image.open("huge_photo.jpg")
img = img.resize((800, 600))  # Scale first
# Now work with smaller image
    \end{lstlisting}
\end{frame}
